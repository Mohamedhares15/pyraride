
model Notification {
  id        String   @id @default(uuid())
  userId    String
  type      String   // "booking_confirmed", "booking_completed", "new_message", "review_received", etc.
  title     String
  message   String   @db.Text
  data      Json?    // Additional data (booking ID, user ID, etc.)
  read      Boolean  @default(false)
  createdAt DateTime @default(now())
  
  user User @relation(fields: [userId], references: [id], onDelete: Cascade)
  
  @@index([userId, read])
  @@index([createdAt])
}
